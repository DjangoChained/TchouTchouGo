113 def parse_gtfs_stops_traintype(lines):
114  """Créer des objets TrainType depuis les données du fichier stops.txt du
115  format GTFS."""
116  # L'expression régulière convertit un identifiant de station SNCF du type
117  # StopPoint:OCETrain TER-87576173
118  # en un type de train ("Train TER" dans l'exemple).
119  regex = re.compile(r"^.*OCE(.*)-[0-9]+$", re.MULTILINE)
120  # Une compréhension de liste utilisant lines va trier et ne récupérer que
121  # les ID commençant par "StopPoint" puisque StopArea ne contient pas de
122  # types de trains. Une seconde compréhension de liste va ensuite exécuter
123  # l'expression régulière.
124  # set() permet de supprimer les très nombreux doublons.
125  # On reconvertit ensuite en liste avec list() pour créer des TrainTypes,
126  # dans une troisième compréhension de liste, en évitant les doublons.
127  traintypes = [TrainType.objects.create(name=name) for name in list(set(
128  [regex.sub('\\1', stopid) for stopid in
129  [line[0] for line in lines if line[0].startswith("StopPoint")]]))
130  if not TrainType.objects.filter(name=name).exists()]
131  print("Writing to database...")
132  for tt in traintypes:
133  tt.save()

152 def parse_gtfs_trains(trips, stop_times_path):
153  """Créer des objets Train et Halt correspondant aux données des fichiers
154  trips.txt et stop_times.txt du format GTFS.
155  Attention : Cette méthode prend en paramètre un lecteur de fichiers CSV
156  pour trips.txt et un chemin d'accès seulement pour stop_times.txt."""
157  stop_times_file = open(stop_times_path)
158  stop_times = csv.reader(stop_times_file)
159  # Permet de trouver le type de train
160  tt_regex = re.compile(r"^.*OCE(.*)-[0-9]+$", re.MULTILINE)
161  trains, halts = [], []
162  period_ids = Period.objects.values_list("id", flat=True)
163  for trip in trips:
164  stop_times_file.seek(1)
165  trip_stop_times = [s for s in stop_times if s[0] == trip[2]]
166  if int(trip[1]) in period_ids:
167  p = int(trip[1])
168  else:
169  p = None
170  t = Train.objects.create(
171  number=int(trip[3]), traintype=TrainType.objects.get(
172  name=tt_regex.sub('\\1', trip_stop_times[0][3])))
173  t.period_id = p
174  trains.append(t)
175  halts.extend(parse_gtfs_halts_train(trip[2], trip_stop_times, t.id))
176  print("Writing to database...")
177  for t in trains:
178  t.save()
179  for h in halts:
180  h.save()

183 def parse_gtfs_halts_train(trip_id, stop_times, train_id):
184  """Créer des objets Halt pour un train donné depuis les données du fichier
185  stop_times.txt du format GTFS."""
186  halts = []
187  station_id_regex = re.compile(r"^.*OCE.*-([0-9]+)$", re.MULTILINE)
188  for stop_time in stop_times:
189  ar_time = stop_time[1].split(':', 2)
190  ar_sec = time(hour=int(ar_time[0]) % 24, minute=int(ar_time[1]))
191  dep_time = stop_time[2].split(':', 2)
192  dep_sec = time(hour=int(dep_time[0]) % 24, minute=int(dep_time[1]))
193  h = Halt.objects.create(
194  arrival=ar_sec, departure=dep_sec, sequence=stop_time[4])
195  h.station_id = int(station_id_regex.sub('\\1', stop_time[3]))
196  h.train_id = train_id
197  halts.append(h)
198  return halts
