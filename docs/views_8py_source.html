224 def travel_map_geojson(request, travel_id):
225  from django.db.models import Q
226  stations = Station.objects.filter(
227  Q(halt__ticket_start_set__travel__id=travel_id) |
228  Q(halt__ticket_end_set__travel__id=travel_id)).distinct()
229  data = [geojson.Feature(geometry=geojson.Point((s.lng, s.lat)),
230  properties={"title": s.name})
231  for s in stations]
232  data.append(geojson.Feature(
233  geometry=geojson.LineString([(s.lng, s.lat) for s in stations]),
234  properties={"strokeColor": "red", "strokeWeight": 3}))
235  return HttpResponse(geojson.dumps(geojson.FeatureCollection(data)),
